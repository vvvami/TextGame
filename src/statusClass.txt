public class net.vami.interactables.Status {

    private String name;
    private boolean harmful;
    private boolean stackDuration;
    private boolean stackAmplifier;
    public static final net.vami.interactables.Status CRIPPLED = new net.vami.interactables.Status("Crippled",false, true,true);
    public static final net.vami.interactables.Status BURNING = new net.vami.interactables.Status("Burning",true, true,true);
    public static final net.vami.interactables.Status FROZEN = new net.vami.interactables.Status("Frozen",false,true,true);
    public static final net.vami.interactables.Status PARALYZED = new net.vami.interactables.Status("Paralyzed",false, true, true);
    public static final net.vami.interactables.Status BLESSED = new net.vami.interactables.Status("Blessed",true, true, false);
    public static final net.vami.interactables.Status IMMUNE = new net.vami.interactables.Status("Immune",false, true, false);
    public static final net.vami.interactables.Status INSPIRED = new net.vami.interactables.Status("Inpired",true, false, false);
    public static final net.vami.interactables.Status BLEEDING = new net.vami.interactables.Status("Bleeding",true,true,true);

    public net.vami.interactables.Status() {
    }

    public net.vami.interactables.Status(String name, boolean stackAmplifier, boolean stackDuration, boolean harmful) {
        this.name = name;
        this.stackAmplifier = stackAmplifier;
        this.stackDuration = stackDuration;
        this.harmful = harmful;
    }

    public String getName() {
        return name;
    }

    public boolean stacksAmplifier() {
        return stackAmplifier;
    }

    public boolean stacksDuration() {
        return stackDuration;
    }

    public boolean isHarmful() {
        return harmful;
    }



}
